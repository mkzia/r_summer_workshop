4.2. DataFrames#

DataFrame is a table or a 2D array with rows and columns.

student_ids = c(111, 222, 333)
names = c('John', 'Jane', 'Jack')
grades = c(70, 89, 85)

student_info = data.frame(
    id = student_ids,
    name = names,
    grade = grades
)

4.2.1. Select a single column by name#

student_info['id']
student_info['name']

4.2.2. Select a multiple columns by name#

student_info[c('id', 'name')]

4.2.3. Select a single column by index#

  • [ROW, COLUMN]

student_info[,1]
student_info[,2]

4.2.4. Select a multiple columns by index#

  • [ROW, COLUMN]

student_info[,c(1, 2)]

4.2.5. Select a single row, all columns#

student_info[1,]

4.2.6. Select a single row, some columns#

student_info[1,c(1, 2)]

4.2.7. Select a single row, some columns#

student_info[1,c(1, 2)]
student_info[1,c('id', 'name')]

4.2.8. Select multiple rows by range, some columns#

student_info[1:2,c(1, 3)]

4.2.9. Select multiple rows by index, some columns#

student_info[c(1, 3),c(1, 3)]

4.2.10. Read TSV file#

df = read.csv('students.tsv', sep='\t')

4.2.11. Select All Melissas#

df[df['First'] == 'Melissa']

4.2.12. Add average column#

df['Average'] = round((df$Exam1 + df$Exam2 + df$Exam3)/3)

4.2.13. Get Max Average#

df[which.max(df$Average), ]

4.2.14. Get Min Average#

df[which.max(df$Average), ]

4.2.15. Order column#

df[order(df$Average)]

4.2.16. Select all grades >=70#

gte70 = df[df$Average>=70,]

4.2.17. Load SEPA Data#

df = read.csv('SEPA-2023-Metadata.csv')

4.2.18. Split GPS data#

Value = df[df$Metadata == 'GPS', 'Value']
temp = str_split_fixed(Value, ', ', 2)
Latitude = as.numeric(temp[,1])
Longitude= as.numeric(temp[,2])
GPS = df[df$Metadata == 'GPS', ]
GPS['Latitude'] = Latitude
GPS['Longitude'] = Longitude

4.2.19. Combine Value with Unit#

df['Combined'] = paste(df[,'Value'], df[,'Unit'])
df$Combined = paste(df$Value, df$Unit)
